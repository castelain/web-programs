        function hide(obj) {
            var children = $(obj).children();
            var length = children.length;
            for (var i = 0; i < length; i++) {
                if (children[i].tagName == "IMG") {
                    children[i].style.display = "none";
                }
            }
        }

        function show(obj) {
            var children = $(obj).children();
            var length = children.length;
            for (var i = 0; i < length; i++) {
                if (children[i].tagName == "IMG") {
                    children[i].style.display = "inline";
                }
            }
        }
